import torch
import numpy as np
import sys
sys.path.append("..")
import d2lzh_pytorch as d2l

# 生成人工数据集
# y = 1.2x - 3.4x^2 + 5.6x^3 + 5 + epsilon

n_train, n_test, true_w, true_b = 100, 100, [1.2, 3.4, 5.6], 5
# 生成 200 * 1的二维矩阵
features = torch.randn(n_train + n_test, 1)
# 组合上述的矩阵 形成：[[200*1] [200*1] [200*1]]的矩阵 分别表示 x x^2 x^3 
poly_features = torch.cat((features, torch.pow(features, 2), torch.pow(features, 3)), 1)

# 非线性的函数模型
labels = true_w[0] * poly_features[:, 0] + true_w[1] * poly_features[:, 1] + true_w[2] * poly_features[:, 2] + true_b
# 再加上一个随机扰动epsilon
labels += torch.tensor(np.random.normal(0, 0.01, size=labels.size()), dtype=torch.float)

#直接使用定义在包中的semilogy函数

num_epochs, loss = 100, torch.nn.MSELoss()
def fit_and_plot(train_features, test_features, train_labels, test_labels):
    net = torch.nn.Linear(train_features.shape[-1], 1)
    batch_size = min(10, train_labels.shape[0])
    # 构造数据集 可以将train_features 理解为 x ; train_labels 理解为y
    dataset = torch.utils.data.TensorDataset(train_features, train_labels)
    
    train_iter = torch.utils.data.DataLoader(dataset, batch_size, True)

    # 使用随机梯度下降算法
    optimizer = torch.optim.SGD(net.parameters(), lr=0.01)
    train_ls, test_ls = [], []

    for _ in range(num_epochs):
        for X, y in train_iter:
            l = loss(net(X), y.view(-1, 1))
            optimizer.zero_grad()
            l.backward()
            optimizer.step()
        train_labels = train_labels.view(-1, 1)
        test_labels = test_labels.view(-1, 1)

        # 函数如何知道
        train_ls.append(loss(net(train_features), train_labels).item())
        test_ls.append(loss(net(test_features), test_labels).item())

    print('final epoch: train loss', train_ls[-1], 'test loss', test_ls[-1])
    d2l.semilogy(range(1, num_epochs + 1), train_ls, 'epochs', 'loss', range(1, num_epochs + 1), test_ls, ['train', 'test'])
    print('weight: ', net.weight.data, '\nbias', net.bias.data)

# 多项式函数拟合的结果
fit_and_plot(
    train_features = poly_features[:n_train,:],
